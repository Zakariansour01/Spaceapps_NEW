const deg2rad = d => d * Math.PI / 180;
const rad2deg = r => r * 180 / Math.PI;

// ---------- تحويل تاريخ إلى Julian Date ----------
function dateToJD(dateObj) {
    return dateObj.getTime()/86400000 + 2440587.5;
}

// ---------- حل معادلة كبلر (Newton-Raphson) ----------
function solveKepler(M, e, tol = 1e-8, maxIter = 50) {
    let Mwrap = ((M + Math.PI) % (2*Math.PI)) - Math.PI;
    let E = (e < 0.8) ? Mwrap : Math.PI;
    for (let iter = 0; iter < maxIter; iter++) {
        let f = E - e * Math.sin(E) - Mwrap;
        let fprime = 1 - e * Math.cos(E);
        let dE = f / fprime;
        E = E - dE;
        if (Math.abs(dE) < tol) {
            return E;
        }
    }
    return E;
}

// ---------- من E نحسب x', y' في مستوى المدار ----------
function orbitalPlaneCoords(a, e, E) {
    const cosE = Math.cos(E);
    const sinE = Math.sin(E);
    const r = a * (1 - e * cosE);
    const xprime = a * (cosE - e);
    const yprime = a * Math.sqrt(Math.max(0, 1 - e * e)) * sinE
    return { r, xprime, yprime };
}

// ---------- تحويل من (x',y',0) إلى (xecl, yecl, zecl) حسب ω, Ω, i ----------
function toEcliptic(xp, yp, omega, Omega, i) {
    const cosw = Math.cos(omega), sinw = Math.sin(omega);
    const cosO = Math.cos(Omega), sinO = Math.sin(Omega);
    const cosI = Math.cos(i), sinI = Math.sin(i);

    const xecl = (cosw * cosO - sinw * sinO * cosI) * xp +
                 (-sinw * cosO - cosw * sinO * cosI) * yp;

    const yecl = (cosw * sinO + sinw * cosO * cosI) * xp +
                 (-sinw * sinO + cosw * cosO * cosI) * yp;

    const zecl = (sinw * sinI) * xp + (cosw * sinI) * yp;

    return { xecl, yecl, zecl };
}

// ---------- حساب M = L - varpi (بدرجات) ثم تحويل إلى radians ----------
function computeMeanAnomaly(L_deg, varpi_deg) {
    let Mdeg = L_deg - varpi_deg;
    Mdeg = ((Mdeg % 360) + 360) % 360;
    return deg2rad(Mdeg);
}

// Orbit rotation shape for earth
function solveKepler_Orbit(M, e, tol=1e-6, maxIter=50){
    let E = M;
    for (let i = 0; i < maxIter; i++){
        let f = E - e * Math.sin(E) - M;
        let fp = 1 - e * Math.cos(E);
        let dE = f / fp;
        E = E - dE;
        if(Math.abs(dE) < tol) break;
    }
    return E;
}

function getEarthPosition(time){
    const n = 2 * Math.PI / T;
    const M = n * time;
    const E = solveKepler_Orbit(M, e);

    const x = a * (Math.cos(E) - e);
    const y = a * Math.sqrt(1 - e * e) * Math.sin(E);
    const z = 0;

    return new THREE.Vector3(x, y, z);
}

function getApolloPosition(t) {
    const n = 2 * Math.PI / T_apo;
    const M = n * t;
    const E = solveKepler_Apo(M % (2 * Math.PI), e_apo);
    

    const r = a_apo * (1 - e_apo * Math.cos(E));

    const cosν = (Math.cos(E) - e_apo) / (1 - e_apo * Math.cos(E));
    const sinν = (Math.sqrt(1 - e_apo*e_apo) * Math.sin(E)) / (1 - e_apo * Math.cos(E));
    const ν = Math.atan2(sinν, cosν);

    let pos = new THREE.Vector3(r * Math.cos(ν), r * Math.sin(ν), 0);

    const rotZ_Omega = new THREE.Matrix4().makeRotationZ(Omega_apo);
    const rotX_i = new THREE.Matrix4().makeRotationX(i_apo);
    const rotZ_omega = new THREE.Matrix4().makeRotationZ(omega_apo);

    pos.applyMatrix4(rotZ_omega);
    pos.applyMatrix4(rotX_i);
    pos.applyMatrix4(rotZ_Omega);

    return pos;
}

function createAsteroidWithRealOrbit(name, orbitClass, diameter, orbitalData) {
    if (!orbitalData) return;
    
    const radius = Math.max(diameter * SCALE_FACTOR * 0.5, 3);
    // console.log(radius)
    
    let color, orbitType;
    
    // Determine which orbit to place asteroid on
    if (orbitClass.includes('Apollo') || orbitClass.includes('APO')) {
        color = 'red';
        orbitType = 'apollo';
    } else if (orbitClass.includes('Amor') || orbitClass.includes('AMO')) {
        color = 'orange';
        orbitType = 'amor';
    } else if (orbitClass.includes('Aten') || orbitClass.includes('ATE')) {
        color = 'green';
        orbitType = 'aten';
    } else if (orbitClass.includes('Atira') || orbitClass.includes('IEO') || orbitClass.includes('ATI')) {
        color = 'cyan';
        orbitType = 'atira';
    } else {
        color = 'white';
        orbitType = 'apollo';
    }
    
    console.log(`${name}: ${orbitClass} -> ${orbitType} orbit`);
    
    // Create asteroid mesh
    const asteroidMaterial = new THREE.MeshStandardMaterial({ 
        color: color, 
        metalness: 0.5, 
        roughness: 0.7 
    });
    const asteroidMesh = new THREE.Mesh(
        new THREE.SphereGeometry(radius, 16, 16), 
        asteroidMaterial
    );
    
    scene.add(asteroidMesh);
    


     // Extract orbital period from NASA data (in days)
    const orbitalPeriodDays = parseFloat(orbitalData.orbital_period) || 365;

    // Convert to simulation units (similar to how you defined T for Earth)
    const T_asteroid = orbitalPeriodDays * 24 * 3600; // in seconds
    // Calculate time increment for this specific asteroid
    // This makes it complete one orbit in T_asteroid simulation time
    const asteroidSpeed = (2 * Math.PI / T_asteroid) * Speed_Factor;

    
    // Store asteroid with random starting position on the correct orbit
    const asteroidData = {
        mesh: asteroidMesh,
        name: name,
        time: Math.random() * T_asteroid,
        speed: asteroidSpeed,
        period: T_asteroid,
        orbitType: orbitType,  // Make sure this is included!
        radius: radius,
        diameter: diameter
    };
    
    // Add to appropriate array based on orbit type
    if (orbitType === 'apollo') {
        apolloAsteroids.push(asteroidData);
    } else if (orbitType === 'amor') {
        amorAsteroids.push(asteroidData);
    } else if (orbitType === 'aten') {
        atenAsteroids.push(asteroidData);
    } else if (orbitType === 'atira') {
        atiraAsteroids.push(asteroidData);
    }
    // Add to master list for control panel
    allAsteroids.push(asteroidData);
    // Update dropdown
    updateAsteroidDropdown();
     // خزّن كل شيء داخل asteroidData

    if (!orbitalData) return;
    
}

    function updateOrbitAfterSpeedChange(asteroid, newSpeedKmS) {
    const mu = 1.327e11; // km^3/s^2 ثابت جاذبية الشمس
    const r = asteroid.currentDistanceKm || (asteroid.a * (1 - asteroid.e) * AU_KM); 
    // إذا ما عندك currentDistanceKm خذ الحضيض تقريبياً

    const v = newSpeedKmS; // السرعة الجديدة km/s

    // نصف المحور الأكبر الجديد
    const a_new_km = 1 / ( (2/r) - (v*v/mu) ); 
    const a_new_AU = a_new_km / AU_KM;

    // الزخم الزاوي الجديد
    const h_new = r * v; 

    // الشذوذ الجديد (e)
    const e_new = Math.sqrt( Math.max(0, 1 - (h_new*h_new) / (a_new_km*mu)) );

    // حدّث خصائص الكويكب
    asteroid.a = a_new_AU;
    asteroid.e = e_new;

    console.log(`New orbit for ${asteroid.name}: a=${a_new_AU.toFixed(3)} AU, e=${e_new.toFixed(3)}`);

    // أعد بناء خط المدار
    rebuildOrbitGeometry(asteroid);
}

    function rebuildOrbitGeometry(asteroid) {
    if (!asteroid.orbitLine) {
        console.warn("Asteroid has no orbit line to update:", asteroid.name);
        return;
    }

    const points = [];
    const numPoints = 360;

    for (let k = 0; k <= numPoints; k++) {
        const M = 2 * Math.PI * (k / numPoints); // متوسط الشذوذ
        const E = solveKepler_Apo(M, asteroid.e); // حل معادلة كبلر

        // نصف القطر الشعاعي r
        const r = asteroid.a * AU_KM / AU_KM * (1 - asteroid.e * Math.cos(E)); 
        // (ضربت وقسمت AU_KM فقط لو عندك وحدة بالكيلومتر – تأكد عندك أي وحدة تستعمل)

        // إحداثيات 2D في مستوى المدار
        const cosν = (Math.cos(E) - asteroid.e) / (1 - asteroid.e * Math.cos(E));
        const sinν = (Math.sqrt(1 - asteroid.e*asteroid.e) * Math.sin(E)) / (1 - asteroid.e * Math.cos(E));
        const ν = Math.atan2(sinν, cosν);

        let pos = new THREE.Vector3(r * Math.cos(ν), r * Math.sin(ν), 0);

        // التحويل للإحداثيات ثلاثية الأبعاد
        const rotZ_Omega = new THREE.Matrix4().makeRotationZ(asteroid.Omega || 0);
        const rotX_i = new THREE.Matrix4().makeRotationX(asteroid.i || 0);
        const rotZ_omega = new THREE.Matrix4().makeRotationZ(asteroid.omega || 0);

        pos.applyMatrix4(rotZ_omega);
        pos.applyMatrix4(rotX_i);
        pos.applyMatrix4(rotZ_Omega);

        points.push(pos);
    }

    // تحديث الـ BufferGeometry
    asteroid.orbitLine.geometry.dispose(); // امسح القديم
    asteroid.orbitLine.geometry = new THREE.BufferGeometry().setFromPoints(points);
}

function updateApolloOrbit() {
    apolloAsteroids.forEach(asteroid => {
        // Use individual speed - Kepler handles variable speed within orbit
        asteroid.time += asteroid.speed;

        asteroid.mesh.position.copy(pos);
    });

    const earthPos = getEarthPosition(elapsedTime);

    // Update all asteroids on their respective orbits using Kepler's equation (variable speed)
    updateApolloOrbit();
